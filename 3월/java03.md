### 인덱스 탐색 알고리즘

int index = -1;
String target = "치킨";
for (int i = 0; i < foods.length; i++) {
    if(target.equals(foods[i])){
        index = i;
        break;
    }
}
System.out.println("찾은 인덱스: " + index);
